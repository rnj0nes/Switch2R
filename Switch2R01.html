library(pipeR)
library(xtable)
library(plyr)

data %>>%
  {do.call(data.frame, 
           list(mean = apply(., 2, mean),
                sd = apply(., 2, sd),
                median = apply(., 2, median),
                min = apply(., 2, min),
                max = apply(., 2, max),
                n = apply(., 2, length)))} %>>%
  xtable(caption = "Summary of the data", 
         digits=2) %>>%
  print(type = "html", caption.placement="top",
        html.table.attributes = FALSE)
